        // 處理圖片預覽

        function previewImage(event, previewId, placeholderId) {

            const input = event.target;

            const preview = document.getElementById(previewId);

            const placeholder = document.getElementById(placeholderId);


            if (input.files && input.files[0]) {

                const reader = new FileReader();

                reader.onload = function(e) {

                    preview.src = e.target.result;

                    preview.classList.remove('hidden');

                    placeholder.classList.add('hidden');

                }

                reader.readAsDataURL(input.files[0]);

            }

        }


        // 模擬生成過程

        function generateTryOn() {

            const personImg = document.getElementById('personPreview').src;

            const clothingImg = document.getElementById('clothingPreview').src;


            // 檢查是否兩張圖都已上傳

            if (!personImg || personImg === window.location.href || 

                !clothingImg || clothingImg === window.location.href) {

                alert("請先上傳人物及衣服照片！");

                return;

            }


            const btn = document.getElementById('generateBtn');

            const resultSection = document.getElementById('resultSection');

            const loadingState = document.getElementById('loadingState');

            const finalResult = document.getElementById('finalResult');


            // 顯示載入狀態

            btn.disabled = true;

            btn.innerHTML = '<i class="fa-solid fa-spinner fa-spin mr-2"></i>處理中...';

            resultSection.classList.remove('hidden');

            resultSection.classList.add('flex');

            loadingState.classList.remove('hidden');

            loadingState.classList.add('flex');

            finalResult.classList.add('hidden');


            // 模擬網絡延遲 (3秒後顯示結果)

            setTimeout(() => {

                loadingState.classList.add('hidden');

                loadingState.classList.remove('flex');

                finalResult.classList.remove('hidden');

                

                // 恢復按鈕

                btn.disabled = false;

                btn.innerHTML = '<i class="fa-solid fa-wand-magic mr-2"></i>再次換裝';


                // 畫出模擬合成圖

                drawMockup(personImg, clothingImg);


                // 捲動到結果區

                resultSection.scrollIntoView({ behavior: 'smooth' });

            }, 3000);

        }


        // 簡單的 Canvas 模擬合成 (將衣服疊加在人像上)

        function drawMockup(personSrc, clothingSrc) {

            const canvas = document.getElementById('resultCanvas');

            const ctx = canvas.getContext('2d');

            

            const pImg = new Image();

            const cImg = new Image();

            

            pImg.onload = () => {

                // 設定 canvas 大小與人物照片相同

                canvas.width = pImg.width;

                canvas.height = pImg.height;

                

                // 畫上人物

                ctx.drawImage(pImg, 0, 0, canvas.width, canvas.height);

                

                cImg.onload = () => {

                    // 模擬把衣服放在中間 (這只是一個非常簡化的視覺模擬)

                    const cWidth = canvas.width * 0.6; // 衣服佔據寬度比例

                    const cHeight = cImg.height * (cWidth / cImg.width);

                    const x = (canvas.width - cWidth) / 2;

                    const y = canvas.height * 0.25; // 大約放在胸前位置

                    

                    // 加上一點透明度讓它看起來像合成

                    ctx.globalAlpha = 0.9;

                    ctx.drawImage(cImg, x, y, cWidth, cHeight);

                    ctx.globalAlpha = 1.0;

                };

                cImg.src = clothingSrc;

            };

            pImg.src = personSrc;

        }

        function downloadResult() {

            link.download = 'ai-tryon-result.png';

            link.href = canvas.toDataURL('image/png');

            link.click();

        }
